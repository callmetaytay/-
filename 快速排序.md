# 快速排序

---

**递归时使用j**

```c++
void quick_sort(vector<int> &num, int l, int r) {
    if(l>=r)return ;
	int x = num[(l + r) / 2], i = l - 1, j = r + 1;
	while (i < j) {
		do i++; while (num[i] < x);
		do j--; while (num[j] > x);
		if (i < j)swap(num[i], num[j]);
	}
	quick_sort(num, l, j);
	quick_sort(num, j+1, r);
```

**递归时使用i**

```c++
void quick_sort(vector<int> &num, int l, int r) {
    if(l>=r)return ;
	int x = num[(l + r + 1) / 2]/*注意这里是l+r+1*/, i = l - 1, j = r + 1;
	while (i < j) {
		do i++; while (num[i] < x);
		do j--; while (num[j] > x);
		if (i < j)swap(num[i], num[j]);
	}
	quick_sort(num, l, i-1);
	quick_sort(num, i, r);
```
